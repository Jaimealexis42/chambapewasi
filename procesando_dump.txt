function detectMediaType(uri: string, base64: string): string {
  if (uri) {
    const lower = uri.toLowerCase();
    if (lower.includes('.png')) return 'image/png';
    if (lower.includes('.jpg') || lower.includes('.jpeg')) return 'image/jpeg';
  }
  if (base64) {
    if (base64.startsWith('iVBOR')) return 'image/png';
    if (base64.startsWith('/9j/')) return 'image/jpeg';
  }
  return 'image/jpeg';
}

export default function Procesando({ archivo, onDone, config: configProp }: { archivo: any, onDone: (result: any) => void, config?: any }) {
  const [paso, setPaso] = useState(0);
  const [error, setError] = useState('');
  const cfg = configProp || { ciudad: 'Lima', pisos: 1, tipoObra: 'casa' };

  

  useEffect(() => {
    const interval = setInterval(() => {
    const interval = setInterval(() => {
      setPaso(p => (p < pasos.length - 1 ? p + 1 : p));
    }, 600);

    const analizar = async (intento = 1) => {

      try {
        let messages: any[];

        if (archivo?.tipo === 'imagen' && archivo?.base64) {
          const mediaType = archivo.base64?.startsWith('iVBOR') ? 'image/png' : 'image/jpeg';
          messages = [{
            role: 'user',
            content: [
              { type: 'image', source: { type: 'base64', media_type: mediaType, data: archivo.base64 } },
              { type: 'text', text: getPrompt(cfg) }
            ]
          }];
        } else {
          messages = [{
            role: 'user',
            content: getPrompt(cfg) + '\n\nNota: No se proporciono imagen. Genera un presupuesto ejemplo para vivienda unifamiliar de 120m2 en Lima.',
          }];
        }

        const response = await fetch('https://api.anthropic.com/v1/messages', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
            'x-api-key': API_KEY,
            'anthropic-version': '2023-06-01',
          },
          body: JSON.stringify({
            model: 'claude-opus-4-6',
            max_tokens: 16000,
            messages,
          }),
        });

        if (!response.ok) {
          const err = await response.text();
          throw new Error('API error ' + response.status + ': ' + err);
        }

        const data = await response.json();
        const texto = data.content?.[0]?.text || '';
        const match = texto.match(/\{[\s\S]*\}/);
        const jsonStr = match ? match[0] : texto;
        const resultado = JSON.parse(jsonStr);
        resultado.ciudad = (configProp || { ciudad: 'Lima' }).ciudad;
        clearInterval(interval);
        onDone(resultado);
      } catch (e: any) {
        clearInterval(interval);
        setError(e.message || 'Error desconocido');
      }
    };

    analizar();
    return () => clearInterval(interval);
  }, []);

  if (error) {
    return (
      <View style={styles.container}>
        <Text style={styles.logo}>PresupIA</Text>
        <Text style={styles.errorIcon}>⚠️</Text>
        <Text style={styles.errorTitle}>Error al procesar</Text>
        <Text style={styles.errorMsg}>{error}</Text>
      </View>
    );
  }

  return (
    <View style={styles.container}>
      <Text style={styles.logo}>PresupIA</Text>
      <ActivityIndicator size="large" color="#00C896" style={styles.spinner} />
      <Text style={styles.paso}>{pasos[paso]}</Text>
      <View style={styles.dotsRow}>
        {pasos.map((_, i) => (
          <View key={i} style={[styles.dot, i <= paso && styles.dotActive]} />
        ))}
      </View>
      <Text style={styles.sub}>La IA esta analizando tu plano{'\n'}Esto puede tomar 20-30 segundos</Text>
    </View>
  );
}
